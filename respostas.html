1. Área de um círculo

import math
raio = 0.15
area = math.pi * (raio ** 2)
print(f"Área do círculo: {area:.4f} m²")

2. Conversão de tempo

total = 3670
horas = total // 3600
resto = total % 3600
minutos = resto // 60
segundos = resto % 60
print(horas, 'h', minutos, 'm', segundos, 's')

3. Pseudocódigo

ler a, b, c
maior = a
se b > maior então
  maior = b
fimse
se c > maior então
  maior = c
fimse
imprimir maior
